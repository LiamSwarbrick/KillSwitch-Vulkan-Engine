Vertex pulling:
We will have different types of meshes with different vertex layouts.
For instance, skinned vs nonskinned meshes, the skinned meshes
will also have vertex weights.

Instead of managing the different vertex attribute formats
like we normally do, 'vertex pulling' involves just
having your vertex data in a big opaque buffer, which you index
in the vertex shader by gl_VertexID.

So a mesh can have an int for the type, and that tells the shader
how to parse that.
E.g.
struct MeshGPU
{
    uint mesh_type;
    uint vertex_offset;  // index into vertex buffer (not bytes)
};
- Push constants or draw-indexed SSBO to pick the mesh
struct VertexStatic
{
    vec3 position;
    vec3 normal;
    vec2 uv;
};
struct VertexSkinned
{
    vec3 position;
    vec3 normal;
    vec2 uv;
    uvec4 joints;
    vec4 weights;
};

Breakdown of vertex pulling:
============================
Instead of this:
- Multiple vertex buffers
- Different VkVertexInputAttributeDescriptions
- Different pipelines for skinned vs non-skinned

We do this:
- One big SSBO (or storage buffer)
- Index into it using gl_VertexID
- Decode based on mesh_type

The CPU side decides:
- which mesh is drawn
- how many vertices
- what mesh_type is
- base offset into the big buffer

The shader decides:
- how to interpret the bytes

EXAMPLE:
```
#version 460

layout(location = 0) out vec3 vNormal;
layout(location = 1) out vec2 vUV;

/* ---- Mesh metadata ---- */

struct Mesh
{
    uint mesh_type;
    uint vertex_offset;
};

layout(std430, set = 0, binding = 0) readonly buffer MeshBuffer
{
    Mesh meshes[];
};

/* ---- Raw vertex buffer ---- */

layout(std430, set = 0, binding = 1) readonly buffer VertexBuffer
{
    uint data[]; // opaque blob
};

layout(push_constant) uniform Push
{
    uint mesh_index;
} push;

void main()
{
    Mesh mesh = meshes[push.mesh_index];

    uint vertex_id = gl_VertexID;
    uint base = mesh.vertex_offset;

    if (mesh.mesh_type == 0)
    {
        // Static mesh
        uint stride = 8; // vec3 + vec3 + vec2 = 8 uints
        uint v = base + vertex_id * stride;

        vec3 position = read_vec3(v + 0);
        vec3 normal   = read_vec3(v + 3);
        vec2 uv       = read_vec2(v + 6);

        gl_Position = vec4(position, 1.0);
        vNormal = normal;
        vUV = uv;
    }
    else if (mesh.mesh_type == 1)
    {
        // Skinned mesh
        uint stride = 16; // static + joints + weights
        uint v = base + vertex_id * stride;

        vec3 position = read_vec3(v + 0);
        vec3 normal   = read_vec3(v + 3);
        vec2 uv       = read_vec2(v + 6);
        uvec4 joints  = read_uvec4(v + 8);
        vec4 weights = read_vec4(v + 12);

        // Skinning would go here

        gl_Position = vec4(position, 1.0);
        vNormal = normal;
        vUV = uv;
    }
}

Pipeline hashing:
Avoiding the combinatorial explosion from manually specifying all the pipelines.
    Without hashing, you end up with:
    “PBR pipeline”
    “PBR pipeline but depth write off”
    “PBR pipeline but alpha blend”
    “Stylized pipeline but outlined”
    “Stylized pipeline but outlined and x-ray”
Pipeline hashing means you specify what state you want in the draw call,
and then the renderer checks if it already has a pipeline for that,
and if not, it makes it and keeps it.
This way, at startup a bunch of new pipelines get automatically made
but then it's all done and dusted.

The shaders with different pipeline states works well with the material and vertex pulling system.

Example:
// The pipeline cache:
unordered_map<uint64_t, PipelineHandle> pipeline_cache;


struct PipelineDesc
{
    ShaderID shader;  // e.g. PBR, STYLISED, OUTLINE, DEPTH_ONLY

    // Fixed-function state
    bool depth_test;
    bool depth_write;
    bool alpha_blend;

    uint8_t cull_mode;
    uint8_t topology;
    uint8_t color_format;
    uint8_t depth_format;

    // Feature bits
    uint32_t feature_mask;
};

// Index hash map with stable hash (xxHash, CityHash, Murmur, etc.):
uint64_t hash_pipeline(const PipelineDesc& desc)
{
    return hash_bytes(&desc, sizeof(desc));
}
PipelineHandle get_pipeline(const PipelineDesc& desc)
{
    uint64_t key = hash_pipeline(desc);

    auto it = pipeline_cache.find(key);
    if (it != pipeline_cache.end())
        return it->second;

    PipelineHandle handle = create_pipeline(desc);
    pipeline_cache[key] = handle;
    return handle;
}


// The vulkan bit gets hidden:
VkGraphicsPipelineCreateInfo info = {};

info.pStages = get_shader_stages(desc.shader);

info.pDepthStencilState.depthTestEnable  = desc.depth_test;
info.pDepthStencilState.depthWriteEnable = desc.depth_write;

info.pRasterizationState.cullMode = to_vk(desc.cull_mode);
info.pColorBlendState = make_blend(desc.alpha_blend);

// formats, topology, etc...

IMPORTANT: Feature masks instead of shader permutations:
BAD: ShaderID::PBR_SKINNED_SHADOWED_FOG_ALPHA
GOOD:
desc.shader = ShaderID::PBR;
desc.feature_mask =
    FEATURE_SKINNED |
    FEATURE_ALPHA_TEST |
    FEATURE_RECEIVE_SHADOWS;

// This can be done with shader specialization constants.
layout(constant_id = 0) const bool SKINNED = false;
layout(constant_id = 1) const bool ALPHA_TEST = false;
